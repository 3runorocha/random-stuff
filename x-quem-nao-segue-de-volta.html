Quem não te segue de volta

Vale pro X (Twitter). O export oficial do X só entrega número de conta, sem nome, então o jeito mais prático de ter @username de verdade é capturar a lista pelo console. Abra o passo a passo abaixo, gere os dois JSON e jogue aqui.

Como pegar as listas (1 minuto)
  1. Abra x.com/SEU_USUARIO/following e pressione F12 → aba Console.
  2. Cole o script abaixo e dê Enter.
  3. Role a lista até o fim (a captura acontece sozinha conforme carrega).
  4. Rode baixarLista() (ou copiarLista() se o download não abrir). Salve como following.
  5. Vá em x.com/SEU_USUARIO/followers, dê F5, cole o script de novo, role até o fim e rode baixarLista(). Salve como followers.
  6. Suba os dois aqui embaixo e clique em Comparar.
(() => {
  const u = new Map();
  const seen = new Set();
  let last = -1;
  const log = () => { if (u.size !== last) { last = u.size; console.log("Capturados: " + u.size); } };
  const walk = (o) => {
    if (!o || typeof o !== "object") return;
    if (Array.isArray(o)) return o.forEach(walk);
    const lg = (o.legacy && typeof o.legacy === "object") ? o.legacy : o;
    const sn = lg.screen_name || o.screen_name;
    if (sn && typeof sn === "string") u.set(sn.toLowerCase(), { username: sn, name: lg.name || "" });
    for (const k in o) walk(o[k]);
  };
  const handle = (url, text) => {
    if (!url) return;
    if (/graphql/.test(url)) seen.add(url.split("?")[0]);
    if (!/Following|Followers/.test(url)) return;
    try { walk(JSON.parse(text)); log(); } catch (e) {}
  };
  const of = window.fetch;
  window.fetch = async (...a) => {
    const r = await of(...a);
    try { const url = typeof a[0] === "string" ? a[0] : (a[0] && a[0].url); r.clone().text().then(t => handle(url, t)).catch(() => {}); } catch (e) {}
    return r;
  };
  const oo = XMLHttpRequest.prototype.open;
  XMLHttpRequest.prototype.open = function (m, url) {
    this.__u = url;
    this.addEventListener("load", function () { try { handle(this.__u, this.responseText); } catch (e) {} });
    return oo.apply(this, arguments);
  };
  window.baixarLista = () => {
    const arr = [...u.values()];
    try { const b = new Blob([JSON.stringify(arr, null, 2)], { type: "application/json" }); const a = document.createElement("a"); a.href = URL.createObjectURL(b); a.download = "x-lista.json"; a.click(); } catch (e) {}
    console.log("Total: " + arr.length + ". Se o download nao abrir, rode copiarLista()");
    return arr.length;
  };
  window.copiarLista = () => { navigator.clipboard.writeText(JSON.stringify([...u.values()], null, 2)).then(() => console.log("Copiado " + u.size + " perfis. Cole no campo do app.")); };
  window.__urls = () => console.log([...seen].join("\n") || "nenhuma chamada graphql vista ainda");
  console.log("OK (fetch + XHR). Role a lista ate o fim; o contador vai subir. Depois rode baixarLista() ou copiarLista().");
})();
